        function setupPortfolio() {
            document.title = `${USER_DATA.fullName} | Developer Portfolio`;
            document.getElementById('nav-logo').innerText = USER_DATA.fullName.toUpperCase();
            document.getElementById('hero-name').innerText = USER_DATA.fullName;

            document.getElementById('footer-email').href = `mailto:${USER_DATA.contactEmail}`;
            document.getElementById('footer-github').href = USER_DATA.githubUrl;
            document.getElementById('footer-linkedin').href = USER_DATA.linkedinUrl;

            // Render Capabilities Cards Grid
            const skillsContainer = document.getElementById('skills-container');
            USER_DATA.skills.forEach(skill => {
                skillsContainer.innerHTML += `
                    <div class="glass-card p-6 rounded-xl flex items-start space-x-4">
                        <div class="text-xl mt-1 ${skill.color}"><i class="${skill.icon}"></i></div>
                        <div>
                            <h3 class="font-bold text-white mb-0.5">${skill.name}</h3>
                            <p class="text-xs text-gray-400 leading-relaxed">${skill.details}</p>
                        </div>
                    </div>
                `;
            });

            // Render Certifications Section Layout
            const certsContainer = document.getElementById('certs-container');
            USER_DATA.certifications.forEach(cert => {
                certsContainer.innerHTML += `
                    <div class="glass-card p-5 rounded-xl space-y-1">
                        <div class="text-purple-400 text-xs font-semibold tracking-wider flex justify-between items-center">
                            <span>${cert.issuer}</span>
                            <span class="text-[10px] text-gray-500">${cert.date}</span>
                        </div>
                        <h3 class="font-bold text-sm text-white">${cert.title}</h3>
                    </div>
                `;
            });

            // Render Project Portfolio Display Showcase Layout Loops
            const projectsContainer = document.getElementById('projects-container');
            USER_DATA.projects.forEach(project => {
                const tagsList = project.tags.map(t => `<span class="text-[10px] bg-white/5 text-gray-400 border border-white/10 px-2 py-0.5 rounded">${t}</span>`).join('');
                
                // Fallback rendering check to handle incomplete project links cleanly
                const linkMarkup = (project.githubLink && project.githubLink !== '#') 
                    ? `<a href="${project.githubLink}" target="_blank" class="inline-flex items-center text-xs text-purple-400 hover:text-purple-300 font-medium space-x-1">

                            <span>&rarr;</span>
                       </a>`

                            ${linkMarkup}
                        </div>
                    </div>
                `;
